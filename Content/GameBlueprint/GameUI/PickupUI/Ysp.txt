n=len(str)
count=0
i=0
j=n-1
while(i<j):
   if(str[i]!=str[j]):
     count+=1
   i+=1
   j-=1     
     
if(count==0):
  print("Entered string is a Palindrome")
else:
    print("Entered string is not a Palindrome")
